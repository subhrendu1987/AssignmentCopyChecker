int main()
{
    int num;
    cout<<"enter a number:";
    cin>>num;
    int Flag;
    for(int i=num-1;i>=2;i--)
    {
        Flag=0;
        for(int j=2;j<=i/2;j++)
        {
            if(i%j==0){
                Flag=1;
                break;
            }
        }

        {

        }    
    }

#include <iostream>
using namespace std;
int factorial(int n)
{
    if(n > 1)
        return n * factorial(n - 1);
    else
        return 1;
}
int main()
{
    int n;
    cout << "Enter a positive integer: ";
    cin >> n;
    cout << "Factorial of " << n << " = " << factorial(n);
    return 0;
}
